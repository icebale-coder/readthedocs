# Python операции/условия

## Условный оператор if
```bash
In [4]: if a == 5:
   ...:     print('#'*40)
   ...:     print('a равно 5')
   ...: elif a == 6:
   ...:     print('='*40)
   ...:     print('а равно 6')
   ...: else:
   ...:     print ('no five, no six')
   ...:
########################################
a равно 5

In [5]: a = 5

In [6]: a = 6

In [7]: if a == 5:
   ...:     print('#'*40)
   ...:     print('a равно 5')
   ...: elif a == 6:
   ...:     print('='*40)
   ...:     print('а равно 6')
   ...: else:
   ...:     print ('no five, no six')
   ...:
========================================
а равно 6

In [8]: a = 7

In [9]: if a == 5:
   ...:     print('#'*40)
   ...:     print('a равно 5')
   ...: elif a == 6:
   ...:     print('='*40)
   ...:     print('а равно 6')
   ...: else:
   ...:     print ('no five, no six')
   ...:
no five, no six
```
